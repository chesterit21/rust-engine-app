    <script>
        // ============================================================================
        // SECURE AUTHENTICATION SYSTEM - CLIENT-SIDE VALIDATION
        // ============================================================================

        // Configuration
        const CONFIG = {
            API_BASE_URL: '/api/v1',
            MIN_PASSWORD_LENGTH: 8,
            MAX_PASSWORD_LENGTH: 128,
            MAX_EMAIL_LENGTH: 255,
            MAX_NAME_LENGTH: 100,
            MIN_NAME_LENGTH: 2,
            OAUTH_PROVIDERS: {
                google: '/api/v1/oauth/google/authorize',
                facebook: '/api/v1/oauth/facebook/authorize',
                microsoft: '/api/v1/oauth/microsoft/authorize'
            }
        };

        // Security Utilities
        const SecurityUtils = {
            // XSS Prevention
            sanitizeInput(input) {
                const div = document.createElement('div');
                div.textContent = input;
                return div.innerHTML;
            },

            // SQL Injection Pattern Detection
            checkSQLInjection(input) {
                const sqlPattern = /(\b(SELECT|INSERT|UPDATE|DELETE|DROP|CREATE|ALTER|EXEC|EXECUTE)\b)|(--)|(;)/gi;
                return sqlPattern.test(input);
            },

            // XSS Pattern Detection
            checkXSS(input) {
                const xssPattern = /(<script|<iframe|javascript:|onerror=|onclick=)/gi;
                return xssPattern.test(input);
            },

            // Email Validation (RFC 5322 compliant)
            validateEmail(email) {
                const emailRegex = /^[a-zA-Z0-9.!#$%&'*+\/=?^_`{|}~-]+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$/;
                return emailRegex.test(email);
            },

            // Password Strength Validation
            validatePasswordStrength(password) {
                const checks = {
                    length: password.length >= CONFIG.MIN_PASSWORD_LENGTH && password.length <= CONFIG.MAX_PASSWORD_LENGTH,
                    uppercase: /[A-Z]/.test(password),
                    lowercase: /[a-z]/.test(password),
                    number: /[0-9]/.test(password),
                    special: /[^A-Za-z0-9]/.test(password)
                };

                const passed = Object.values(checks).filter(Boolean).length;
                
                return {
                    isValid: checks.length && checks.uppercase && checks.lowercase && checks.number && checks.special,
                    strength: passed >= 5 ? 'strong' : passed >= 3 ? 'medium' : 'weak',
                    checks
                };
            },

            // Name Validation
            validateName(name) {
                // Only allow letters, spaces, hyphens, apostrophes
                const nameRegex = /^[a-zA-Z\s\-']+$/;
                return nameRegex.test(name) && 
                       name.length >= CONFIG.MIN_NAME_LENGTH && 
                       name.length <= CONFIG.MAX_NAME_LENGTH;
            }
        };

        // Form Validation
        const FormValidator = {
            validateLoginForm(formData) {
                const errors = {};

                // Email validation
                if (!formData.email || formData.email.trim() === '') {
                    errors.email = 'Email is required';
                } else if (formData.email.length > CONFIG.MAX_EMAIL_LENGTH) {
                    errors.email = 'Email is too long (max 255 characters)';
                } else if (!SecurityUtils.validateEmail(formData.email)) {
                    errors.email = 'Please enter a valid email address';
                } else if (SecurityUtils.checkSQLInjection(formData.email) || SecurityUtils.checkXSS(formData.email)) {
                    errors.email = 'Invalid characters detected';
                }

                // Password validation
                if (!formData.password || formData.password === '') {
                    errors.password = 'Password is required';
                } else if (formData.password.length < CONFIG.MIN_PASSWORD_LENGTH) {
                    errors.password = `Password must be at least ${CONFIG.MIN_PASSWORD_LENGTH} characters`;
                } else if (formData.password.length > CONFIG.MAX_PASSWORD_LENGTH) {
                    errors.password = `Password is too long (max ${CONFIG.MAX_PASSWORD_LENGTH} characters)`;
                }

                return errors;
            },

            validateRegisterForm(formData) {
                const errors = {};

                // Name validation
                if (!formData.name || formData.name.trim() === '') {
                    errors.name = 'Full name is required';
                } else if (formData.name.trim().length < CONFIG.MIN_NAME_LENGTH) {
                    errors.name = `Name must be at least ${CONFIG.MIN_NAME_LENGTH} characters`;
                } else if (formData.name.length > CONFIG.MAX_NAME_LENGTH) {
                    errors.name = `Name is too long (max ${CONFIG.MAX_NAME_LENGTH} characters)`;
                } else if (!SecurityUtils.validateName(formData.name)) {
                    errors.name = 'Name can only contain letters, spaces, hyphens, and apostrophes';
                } else if (SecurityUtils.checkSQLInjection(formData.name) || SecurityUtils.checkXSS(formData.name)) {
                    errors.name = 'Invalid characters detected';
                }

                // Email validation
                if (!formData.email || formData.email.trim() === '') {
                    errors.email = 'Email is required';
                } else if (formData.email.length > CONFIG.MAX_EMAIL_LENGTH) {
                    errors.email = 'Email is too long (max 255 characters)';
                } else if (!SecurityUtils.validateEmail(formData.email)) {
                    errors.email = 'Please enter a valid email address';
                } else if (SecurityUtils.checkSQLInjection(formData.email) || SecurityUtils.checkXSS(formData.email)) {
                    errors.email = 'Invalid characters detected';
                }

                // Password validation
                if (!formData.password || formData.password === '') {
                    errors.password = 'Password is required';
                } else {
                    const strength = SecurityUtils.validatePasswordStrength(formData.password);
                    if (!strength.isValid) {
                        errors.password = 'Password must contain uppercase, lowercase, number, and special character';
                    }
                }

                // Confirm password validation
                if (!formData.confirmPassword || formData.confirmPassword === '') {
                    errors.confirmPassword = 'Please confirm your password';
                } else if (formData.password !== formData.confirmPassword) {
                    errors.confirmPassword = 'Passwords do not match';
                }

                return errors;
            },

            showErrors(errors, formType) {
                // Clear previous errors
                document.querySelectorAll('.form-input.error').forEach(input => {
                    input.classList.remove('error');
                });
                document.querySelectorAll('.error-message.show').forEach(msg => {
                    msg.classList.remove('show');
                });

                // Show new errors
                Object.keys(errors).forEach(field => {
                    const prefix = formType === 'login' ? 'login' : 'register';
                    const fieldName = field.charAt(0).toUpperCase() + field.slice(1);
                    const input = document.getElementById(`${prefix}${fieldName}`);
                    const errorMsg = document.getElementById(`${prefix}${fieldName}Error`);

                    if (input && errorMsg) {
                        input.classList.add('error');
                        errorMsg.querySelector('span').textContent = errors[field];
                        errorMsg.classList.add('show');
                    }
                });
            },

            clearErrors() {
                document.querySelectorAll('.form-input.error').forEach(input => {
                    input.classList.remove('error');
                });
                document.querySelectorAll('.error-message.show').forEach(msg => {
                    msg.classList.remove('show');
                });
            }
        };

        // API Service
        const APIService = {
            async login(email, password, remember) {
                try {
                    const response = await fetch(`${CONFIG.API_BASE_URL}/auth/login`, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                            'X-Requested-With': 'XMLHttpRequest'
                        },
                        credentials: 'include',
                        body: JSON.stringify({
                            email: SecurityUtils.sanitizeInput(email.trim()),
                            password: password,
                            remember: remember
                        })
                    });

                    const data = await response.json();

                    if (!response.ok) {
                        throw new Error(data.error?.message || 'Login failed');
                    }

                    return data;
                } catch (error) {
                    console.error('Login error:', error);
                    throw error;
                }
            },

                            email: SecurityUtils.sanitizeInput(email.trim()),
                            password: password

                    });

                        throw new Error(data.error?.message || 'Registration failed');
                    }

                    return data;
                } catch (error) {
                    console.error('Registration error:', error);
                    throw error;
                }
            }
